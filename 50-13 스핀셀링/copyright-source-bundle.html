 42  function _getItem(key) {
 43    // 1차: localStorage 시도
 44    try {
 45      const v = localStorage.getItem(key);
 46      if (v) return v;
 47    } catch(e) {}
 48    // 2차: window.name 저장소
 49    return _store[key] || null;
 50  }

 52  function _setItem(key, value) {
 53    try { localStorage.setItem(key, value); } catch(e) {}
 54    _store[key] = value;
 55    try { window.name = JSON.stringify(_store); } catch(e) {}
 56  }

 59    try { localStorage.removeItem(key); } catch(e) {}
 60    delete _store[key];
 61    try { window.name = JSON.stringify(_store); } catch(e) {}

438      const j = Math.floor(Math.random() * (i + 1));
439      [arr[i], arr[j]] = [arr[j], arr[i]];
440    }

 60  function tokenize(text) {
 61    if (!text) return [];
 62    const tokens = new Set();
 63    // 공백 분리 단어
 64    text.toLowerCase().split(/[\s,.()/\[\]"'?!\-—~+:;]+/).forEach(w => { if (w.length >= 2) tokens.add(w); });
 65    // 2-gram (한국어 대응)
 66    const clean = text.replace(/\s+/g, '');
 67    for (let i = 0; i < clean.length - 1; i++) {
 68      const bg = clean.slice(i, i + 2);
 69      if (/[가-힣A-Za-z]{2}/.test(bg)) tokens.add(bg.toLowerCase());
 70    }
 71    return [...tokens];
 72  }

 82      const chunkTokens = tokenize(chunkText);
 83      const chunkSet = new Set(chunkTokens);

220    function renderMd(text) {
221      return text
222        .replace(/```(\w*)\n([\s\S]*?)```/g, function(m, lang, code) { return '<pre><code>' + code.replace(/</g,'&lt;') + '</code></pre>'; })
223        .replace(/`([^`]+)`/g, '<code>$1</code>')
224        .replace(/\*\*(.*?)\*\*/g, '<strong>$1</strong>')
225        .replace(/\*(.*?)\*/g, '<em>$1</em>')
226        .replace(/^- (.+)/gm, '<li>$1</li>')
227        .replace(/(<li>.*<\/li>)/s, '<ul>$1</ul>')
228        .replace(/\n/g, '<br>');
229    }

280    function cosineSim(a, b) {
281      let dot = 0, na = 0, nb = 0;
282      for (let i = 0; i < a.length; i++) { dot += a[i]*b[i]; na += a[i]*a[i]; nb += b[i]*b[i]; }
283      return dot / (Math.sqrt(na) * Math.sqrt(nb) + 1e-8);
284    }

287      if (_kbEmbeddings) return _kbEmbeddings;
288      if (typeof SPIN_KB === 'undefined') return null;

298        _kbEmbeddings = embeddings;
299        try { localStorage.setItem('spin_kb_embeddings_v1', JSON.stringify(embeddings)); } catch(e) {}

304    async function vectorSearch(query, topK) {
305      const qEmb = await embedText(query);
306      if (!qEmb) return [];
307      const kbEmbs = await ensureKBEmbeddings();
308      if (!kbEmbs || typeof SPIN_KB === 'undefined') return [];
309      const scored = SPIN_KB.map((c, i) => ({ chunk: c, score: kbEmbs[i] ? cosineSim(qEmb, kbEmbs[i]) : 0 }));
310      return scored.sort((a, b) => b.score - a.score).slice(0, topK || 5).map(x => x.chunk);
311    }

313    async function hybridSearch(query) {
314      const kwResults = (typeof searchKB === 'function') ? searchKB(query, 5) : [];
315      let vecResults = [];
316      try { vecResults = await vectorSearch(query, 5); } catch(e) {}
317      const seen = {};
318      const merged = [];
319      [...kwResults, ...vecResults].forEach(c => {
320        if (c && !seen[c.id]) { seen[c.id] = true; merged.push(c); }
321      });
322      return merged.slice(0, 6);
323    }

326    async function callGemini(userMsg) {
327      // RAG: 지식베이스 검색
328      let ragContext = '';
329      try {
330        const chunks = await hybridSearch(userMsg);
331        if (chunks && chunks.length > 0) {
332          ragContext = '\n\n=== 플랫폼 지식베이스 (RAG) ===\n' +
333            chunks.map((c, i) => '[' + (i + 1) + '] ' + c.tag + ':\n' + c.content).join('\n\n') +
334            '\n\n위 참고문서를 우선 활용하여 답변하세요.';
335        }
336      } catch(e) {}
337  
338      // 교육생 개인화 데이터
339      let personalContext = '';
340      try {
341        const userStr = localStorage.getItem('spin_user') || (window._store && window._store.spin_user) || '';
342        if (userStr) {
343          const user = JSON.parse(userStr);
344          if (user && user.name) {
345            const scores = {
346              quiz: user.score_quiz || 0,
347              needs: user.score_needs || 0,
348              scenario: user.score_scenario || 0,
349              practice: user.score_practice || 0,
350              roleplay: user.score_roleplay || 0,
351              fab: user.score_fab || 0
352            };
353            const total = Object.values(scores).reduce((s, v) => s + v, 0);
354            const maxScores = { quiz: 80, needs: 80, scenario: 100, practice: 120, roleplay: 100, fab: 90 };
355  
356            // 약점 분석
357            const weakest = Object.entries(scores).sort((a, b) => (a[1] / maxScores[a[0]]) - (b[1] / maxScores[b[0]]))[0];
358            const strongest = Object.entries(scores).sort((a, b) => (b[1] / maxScores[b[0]]) - (a[1] / maxScores[a[0]]))[0];
359  
360            personalContext = '\n\n=== 현재 교육생 정보 ===\n' +
361              '이름: ' + user.name + '\n' +
362              '팀: ' + (user.team_id || '미정') + '\n' +
363              '점수: 퀴즈 ' + scores.quiz + '/80, 니즈 ' + scores.needs + '/80, 질문구분 ' + scores.scenario + '/100, 질문연습 ' + scores.practice + '/120, 롤플레이 ' + scores.roleplay + '/100, FAB ' + scores.fab + '/90\n' +
364              '총점: ' + total + '/600\n' +
365              '강점: ' + strongest[0] + ' (' + strongest[1] + '점)\n' +
366              '약점: ' + weakest[0] + ' (' + weakest[1] + '점)\n' +
367              '개인화 지침: 이 교육생의 약점(' + weakest[0] + ')을 중심으로 맞춤 코칭을 제공하세요. 이름을 불러주세요.';
368          }
369        }
370      } catch(e) {}
371  
372      const contents = [
373        { role: 'user', parts: [{ text: SPIN_KNOWLEDGE + ragContext + personalContext + '\n\n---\n\n사용자 질문: ' + (chatHistory.length === 0 ? '' : '이전 대화:\n' + chatHistory.slice(-8).map(h => h.role + ': ' + h.content).join('\n') + '\n\n현재 질문: ') + userMsg }] }
374      ];
375  
376      const res = await fetch(GEMINI_URL, {
377        method: 'POST',
378        headers: { 'Content-Type': 'application/json' },
379        body: JSON.stringify({
380          contents: contents,
381          generationConfig: {
382            temperature: 0.4,
383            maxOutputTokens: 1024,
384            topP: 0.9
385          }
386        })
387      });
388  
389      if (!res.ok) throw new Error('API ' + res.status);
390      const data = await res.json();
391      return data.candidates[0].content.parts[0].text;
392    }
